`` sql

SELECT 
    t.student_id,
    t.course_id,

    ROUND(
        (
            SUM(
                (t.student_score / t.category_max_score) * t.weight_percentage
            )
            /
            NULLIF(SUM(t.weight_percentage), 0)
        ) * 100
    , 2) AS final_percentage

FROM (
    SELECT 
        ass.student_id,
        a.course_id,
        ac.category_id,

        SUM(ass.score) AS student_score,
        SUM(a.max_score) AS category_max_score,
        ac.weight_percentage

    FROM assessment_submissions ass

    JOIN assessments a 
        ON a.assessment_id = ass.assessment_id

    JOIN assessment_categories ac 
        ON ac.category_id = a.category_id
    GROUP BY 
        ass.student_id,
        a.course_id,
        ac.category_id
) t

GROUP BY 
    t.student_id,
    t.course_id;
